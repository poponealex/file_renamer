CREATE A NEW DIRECTORY WITH AN UNKNOWN NAME

# input #

/bin/bash -> /bin/bashDIR/bash

# output #

/bin/bashDIR ? exists ? -> False
mkdir /bin/bashDIR
/bin/bashDIR/bash ? exists ? -> False
/bin/bash -> /bin/bashDIR/bash



CREATE A NEW DIRECTORY WITH AN EXISTING FILE'S NAME

/bin/bash -> /bin/bash/bash

# output #

/bin/bash ? exists ? -> False
/bin/bash ? isDir ? -> False
/bin/bashAUX ? exists ? -> False
/bin/bash -> /bin/bashAUX
mkdir /bin/bash
/bin/bash/bash ? exists ? -> False
/bin/bashAUX -> /bin/bash/bash
